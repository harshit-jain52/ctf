    ```bash
    echo "<text>" | base64 --encode
    ```

3. ROT13 using Python script:

    ```python
    def shift_letter(c, shift):
        if 'a' <= c <= 'z':
            return chr((ord(c) - ord('a') + shift) % 26 + ord('a'))
        elif 'A' <= c <= 'Z':
            return chr((ord(c) - ord('A') + shift) % 26 + ord('A'))
        else:
            return c

    def shift_text(text, shift):
        return ''.join(shift_letter(c, shift) for c in text)
    ```
